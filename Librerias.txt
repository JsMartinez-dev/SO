#include <stddef.h> -> tipos y macros estándar usados en otras librerías. size_t
#include <sys/types.h> ->tipos de datos usados en el sistema operativo, pid_t, off_t,
#include <sys/wait.h> -> Funcion para manejar procesos hijos , wait(NULL)
#include <unistd.h>  -> Acceso directo a llamadas del sistema POSIX, fork() exec() getpid()
